1.Find the Smaller Angle

    var Minimal_Angle = (h, m) => 
    {
     let hrs = (h*360)/(12)+ (m*360)/ (12*60);
     let min = (m*360)/(60);
          
     let angle = Math.abs(hrs - min);
     if(angle>180){
      angle = 360-angle;
     }
    return angle;
    };

        var Perfect_Check = (N) => 
        {
          var temp =1;
          for(var i=1; i<=N/2;i++){
            if(N%i===0){
              temp=temp+i;
            }
          }
          if(temp===N && temp!==0){
            return "YES"
          }
          else{
            return "NO"
          }
                
        };
