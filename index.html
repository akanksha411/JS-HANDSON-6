3.Perfect Number Check.

        var Perfect_Check = (N) => 
        {
          var temp =1;
          for(var i=1; i<=N/2;i++){
            if(N%i===0){
              temp=temp+i;
            }
          }
          if(temp===N && temp!==0){
            return "YES"
          }
          else{
            return "NO"
          }
                
        };
